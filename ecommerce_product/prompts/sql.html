Partitioning Strategy
  • orders → RANGE on created_at (monthly)
  • analytics_events → RANGE on event_time (daily)
  • notifications → LIST on user region
  • audit_logs → RANGE on logged_at (weekly)
  • inventory_transactions → RANGE on txn_date
⚡
Sharding Keys
  • users → shard by user_id (mod N)
  • products → shard by category_id
  • orders → shard by user_id
  • inventory → shard by warehouse_id
  • payments → shard by merchant_id
📦
Database Engines
  • PostgreSQL — transactional core
  • Cassandra — events, analytics, logs
  • Redis — sessions, carts, rate-limits
  • Elasticsearch — search, catalog
  • ClickHouse — OLAP / reporting
🔁
Replication
  • 1 Primary + 3 Read-replicas per shard
  • Multi-region active-active for carts
  • CDC via Debezium → Kafka
  • Async replication to data warehouse
  • Point-in-time recovery (PITR) 30 days
🔐
Security
  • Column-level encryption (PII, PCI)
  • Row-level security (RLS) policies
  • VPC-isolated DB clusters
  • Secrets rotation via Vault/KMS
  • Full audit trail on sensitive tables

users
Core user identity — sharded by user_id, encrypted PII columns
PostgreSQL
-- ============================================================
-- DOMAIN: USERS & AUTHENTICATION
-- Sharded by user_id, PII columns encrypted at rest (AES-256)
-- ============================================================

CREATE TABLE users (
    user_id          UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    email            TEXT           NOT NULL UNIQUE,       -- encrypted
    email_hash       BYTEA          NOT NULL,              -- SHA-256 for lookup
    phone            TEXT,                                  -- encrypted
    phone_hash       BYTEA,
    full_name        TEXT           NOT NULL,              -- encrypted
    display_name     VARCHAR(80),
    avatar_url       TEXT,
    date_of_birth    DATE,                                  -- encrypted
    gender           VARCHAR(20),
    locale           VARCHAR(10)    NOT NULL DEFAULT 'en-IN',
    timezone         VARCHAR(50)    NOT NULL DEFAULT 'Asia/Kolkata',
    currency_code    CHAR(3)        NOT NULL DEFAULT 'INR',
    country_code     CHAR(2)        NOT NULL DEFAULT 'IN',
    account_type     VARCHAR(20)    NOT NULL DEFAULT 'customer',
                                      -- customer | seller | admin | staff
    status           VARCHAR(20)    NOT NULL DEFAULT 'active',
                                      -- active | suspended | deleted | pending_verify
    tier             VARCHAR(20)    NOT NULL DEFAULT 'standard',
                                      -- standard | silver | gold | platinum | prime
    loyalty_points   BIGINT         NOT NULL DEFAULT 0,
    referral_code    VARCHAR(20)    UNIQUE,
    referred_by      UUID           REFERENCES users(user_id),
    email_verified   BOOLEAN        NOT NULL DEFAULT FALSE,
    phone_verified   BOOLEAN        NOT NULL DEFAULT FALSE,
    kyc_status       VARCHAR(20)    DEFAULT 'unverified',
    last_login_at    TIMESTAMPTZ,
    last_login_ip    INET,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now(),
    updated_at       TIMESTAMPTZ    NOT NULL DEFAULT now(),
    deleted_at       TIMESTAMPTZ                             -- soft delete
);

CREATE INDEX idx_users_email_hash     ON users(email_hash);
CREATE INDEX idx_users_phone_hash     ON users(phone_hash);
CREATE INDEX idx_users_referral       ON users(referral_code);
CREATE INDEX idx_users_status_tier    ON users(status, tier);
CREATE INDEX idx_users_created        ON users(created_at DESC);


CREATE TABLE user_credentials (
    credential_id    UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id          UUID           NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    credential_type  VARCHAR(20)    NOT NULL,   -- password | google | apple | facebook | otp
    identifier       TEXT           NOT NULL,
    secret_hash      TEXT,                       -- Argon2id hash
    salt             BYTEA,
    mfa_secret       TEXT,                       -- TOTP seed, encrypted
    mfa_enabled      BOOLEAN        DEFAULT FALSE,
    last_used_at     TIMESTAMPTZ,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now(),
    UNIQUE(user_id, credential_type)
);

CREATE INDEX idx_creds_user          ON user_credentials(user_id);


CREATE TABLE user_sessions (
    session_id       UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id          UUID           NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    device_id        UUID,
    access_token_jti TEXT           NOT NULL UNIQUE,
    refresh_token    TEXT           NOT NULL,
    ip_address       INET,
    user_agent       TEXT,
    device_type      VARCHAR(20),   -- web | ios | android | desktop
    country_code     CHAR(2),
    is_active        BOOLEAN        NOT NULL DEFAULT TRUE,
    expires_at       TIMESTAMPTZ    NOT NULL,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now(),
    revoked_at       TIMESTAMPTZ
);

CREATE INDEX idx_sessions_user        ON user_sessions(user_id, is_active);
CREATE INDEX idx_sessions_expires     ON user_sessions(expires_at);


CREATE TABLE user_addresses (
    address_id       UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id          UUID           NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    label            VARCHAR(50),   -- Home | Work | Other
    full_name        TEXT           NOT NULL,
    phone            TEXT           NOT NULL,
    address_line1    TEXT           NOT NULL,
    address_line2    TEXT,
    city             VARCHAR(100)   NOT NULL,
    state            VARCHAR(100)   NOT NULL,
    postal_code      VARCHAR(20)    NOT NULL,
    country_code     CHAR(2)        NOT NULL DEFAULT 'IN',
    latitude         DECIMAL(9,6),
    longitude        DECIMAL(9,6),
    geohash          VARCHAR(12),
    is_default       BOOLEAN        NOT NULL DEFAULT FALSE,
    is_verified      BOOLEAN        NOT NULL DEFAULT FALSE,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now()
);

CREATE INDEX idx_addr_user            ON user_addresses(user_id, is_default);
CREATE INDEX idx_addr_geo             ON user_addresses(geohash);


CREATE TABLE user_devices (
    device_id        UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id          UUID           NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    device_fingerprint TEXT         NOT NULL,
    platform         VARCHAR(20),   -- ios | android | web
    os_version       VARCHAR(20),
    app_version      VARCHAR(20),
    push_token       TEXT,
    is_trusted       BOOLEAN        DEFAULT FALSE,
    last_seen_at     TIMESTAMPTZ,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now()
);

CREATE INDEX idx_devices_user         ON user_devices(user_id);
CREATE INDEX idx_devices_push         ON user_devices(push_token) WHERE push_token IS NOT NULL;
Product Catalog Domain
Hierarchical categories, flexible attribute model, multi-image, pricing, SEO
PostgreSQL
-- ============================================================
-- DOMAIN: PRODUCT CATALOG
-- Supports 100M+ SKUs, multi-tenant sellers, EAV attributes
-- ============================================================

CREATE TABLE categories (
    category_id      UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    parent_id        UUID           REFERENCES categories(category_id),
    name             VARCHAR(200)   NOT NULL,
    slug             VARCHAR(200)   NOT NULL UNIQUE,
    description      TEXT,
    image_url        TEXT,
    icon_url         TEXT,
    meta_title       VARCHAR(160),
    meta_description VARCHAR(320),
    depth_level      SMALLINT       NOT NULL DEFAULT 0,
    path             TEXT,          -- materialized path e.g. /electronics/phones/android
    display_order    INTEGER        NOT NULL DEFAULT 0,
    is_active        BOOLEAN        NOT NULL DEFAULT TRUE,
    commission_pct   DECIMAL(5,2)   DEFAULT 0,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now()
);

CREATE INDEX idx_cat_parent           ON categories(parent_id);
CREATE INDEX idx_cat_path             ON categories USING gist(path gist_trgm_ops);
CREATE INDEX idx_cat_slug             ON categories(slug);


CREATE TABLE brands (
    brand_id         UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    name             VARCHAR(200)   NOT NULL UNIQUE,
    slug             VARCHAR(200)   NOT NULL UNIQUE,
    logo_url         TEXT,
    banner_url       TEXT,
    description      TEXT,
    country_of_origin CHAR(2),
    is_verified      BOOLEAN        DEFAULT FALSE,
    is_active        BOOLEAN        DEFAULT TRUE,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now()
);


CREATE TABLE products (
    product_id       UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    seller_id        UUID           NOT NULL REFERENCES sellers(seller_id),
    brand_id         UUID           REFERENCES brands(brand_id),
    category_id      UUID           NOT NULL REFERENCES categories(category_id),
    title            TEXT           NOT NULL,
    slug             TEXT           NOT NULL UNIQUE,
    description      TEXT,
    short_description TEXT,
    sku_prefix       VARCHAR(50)    NOT NULL,
    barcode          VARCHAR(50),
    product_type     VARCHAR(30)    NOT NULL DEFAULT 'physical',
                                     -- physical | digital | service | subscription
    status           VARCHAR(30)    NOT NULL DEFAULT 'draft',
                                     -- draft | pending_review | active | paused | archived
    base_price       NUMERIC(14,2)  NOT NULL,
    compare_price    NUMERIC(14,2),
    cost_price       NUMERIC(14,2),
    currency_code    CHAR(3)        NOT NULL DEFAULT 'INR',
    tax_class        VARCHAR(50),
    hsn_code         VARCHAR(20),   -- for GST
    gst_rate         DECIMAL(5,2),
    avg_rating       DECIMAL(3,2)   DEFAULT 0,
    review_count     INTEGER        DEFAULT 0,
    sold_count       BIGINT         DEFAULT 0,
    view_count       BIGINT         DEFAULT 0,
    wishlist_count   INTEGER        DEFAULT 0,
    is_featured      BOOLEAN        DEFAULT FALSE,
    is_bestseller    BOOLEAN        DEFAULT FALSE,
    meta_title       VARCHAR(160),
    meta_description VARCHAR(320),
    search_vector    TSVECTOR,      -- full-text search
    tags             TEXT[],
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now(),
    updated_at       TIMESTAMPTZ    NOT NULL DEFAULT now(),
    published_at     TIMESTAMPTZ
);

CREATE INDEX idx_prod_category        ON products(category_id, status);
CREATE INDEX idx_prod_seller          ON products(seller_id, status);
CREATE INDEX idx_prod_brand           ON products(brand_id);
CREATE INDEX idx_prod_status          ON products(status, created_at DESC);
CREATE INDEX idx_prod_fts             ON products USING gin(search_vector);
CREATE INDEX idx_prod_tags            ON products USING gin(tags);
CREATE INDEX idx_prod_rating          ON products(avg_rating DESC, sold_count DESC);

-- Auto-update FTS vector
CREATE TRIGGER trg_products_search_vector
  BEFORE INSERT OR UPDATE ON products
  FOR EACH ROW EXECUTE FUNCTION
  tsvector_update_trigger(search_vector, 'pg_catalog.english', title, description, tags);


CREATE TABLE product_variants (

    option1_value    VARCHAR(100),  -- Red
    option2_name     VARCHAR(50),   -- Size
    option2_value    VARCHAR(100),  -- XL
    option3_name     VARCHAR(50),
    option3_value    VARCHAR(100),
    price            NUMERIC(14,2)  NOT NULL,
    compare_price    NUMERIC(14,2),
    cost_price       NUMERIC(14,2),
    weight_grams     INTEGER,
    length_mm        INTEGER,
    width_mm         INTEGER,
    height_mm        INTEGER,
    barcode          VARCHAR(50),
    is_active        BOOLEAN        NOT NULL DEFAULT TRUE,
    position         SMALLINT       DEFAULT 0,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now()
);

CREATE INDEX idx_variant_product      ON product_variants(product_id, is_active);
CREATE INDEX idx_variant_sku          ON product_variants(sku);


CREATE TABLE product_images (

Inventory & Warehouse Domain
Multi-warehouse, reservation ledger, real-time stock tracking
PostgreSQL
-- ============================================================
-- DOMAIN: INVENTORY & WAREHOUSING
-- Ledger-based stock model prevents overselling at scale
-- ============================================================

CREATE TABLE warehouses (
    warehouse_id     UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    name             VARCHAR(200)   NOT NULL,
    code             VARCHAR(20)    NOT NULL UNIQUE,
    warehouse_type   VARCHAR(30)    NOT NULL,   -- fulfillment_center | seller | dark_store | returns_center
    address_line1    TEXT           NOT NULL,
    city             VARCHAR(100)   NOT NULL,
    state            VARCHAR(100)   NOT NULL,
    country_code     CHAR(2)        NOT NULL,
    postal_code      VARCHAR(20)    NOT NULL,
    latitude         DECIMAL(9,6)   NOT NULL,
    longitude        DECIMAL(9,6)   NOT NULL,
    geohash          VARCHAR(12),
    total_area_sqft  INTEGER,
    is_active        BOOLEAN        NOT NULL DEFAULT TRUE,
    operating_hours  JSONB,
    contact_phone    TEXT,
    manager_user_id  UUID,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now()
);

CREATE INDEX idx_wh_geo               ON warehouses(geohash);
CREATE INDEX idx_wh_active            ON warehouses(is_active, country_code);


CREATE TABLE inventory (
    inventory_id     UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    variant_id       UUID           NOT NULL REFERENCES product_variants(variant_id),
    warehouse_id     UUID           NOT NULL REFERENCES warehouses(warehouse_id),
    qty_on_hand      INTEGER        NOT NULL DEFAULT 0,
    qty_reserved     INTEGER        NOT NULL DEFAULT 0,
    qty_available    INTEGER        GENERATED ALWAYS AS (qty_on_hand - qty_reserved) STORED,
    qty_incoming     INTEGER        NOT NULL DEFAULT 0,
    reorder_point    INTEGER        NOT NULL DEFAULT 10,
    reorder_qty      INTEGER        NOT NULL DEFAULT 50,
    bin_location     VARCHAR(50),   -- Aisle-Rack-Shelf e.g. A3-R12-S4
    last_counted_at  TIMESTAMPTZ,
    updated_at       TIMESTAMPTZ    NOT NULL DEFAULT now(),
    UNIQUE(variant_id, warehouse_id),
    CHECK(qty_on_hand >= 0),
    CHECK(qty_reserved >= 0),
    CHECK(qty_on_hand >= qty_reserved)
);

CREATE INDEX idx_inv_variant          ON inventory(variant_id);
CREATE INDEX idx_inv_warehouse        ON inventory(warehouse_id);
CREATE INDEX idx_inv_low_stock        ON inventory(variant_id, warehouse_id)
    WHERE qty_available <= reorder_point;


CREATE TABLE inventory_transactions (
    txn_id           UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    variant_id       UUID           NOT NULL,
    warehouse_id     UUID           NOT NULL,
    txn_type         VARCHAR(30)    NOT NULL,
                     -- purchase_in | sale_out | reservation | reservation_cancel
                     -- return_in | damage_out | adjustment | transfer
    qty_delta        INTEGER        NOT NULL,  -- positive = in, negative = out
    reference_type   VARCHAR(30),
    reference_id     UUID,
    cost_per_unit    NUMERIC(12,4),
    performed_by     UUID,
    notes            TEXT,
    txn_date         DATE           NOT NULL DEFAULT CURRENT_DATE,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now()
) PARTITION BY RANGE (txn_date);

-- Monthly partitions (automated via pg_partman in prod)
CREATE TABLE inventory_transactions_2025_01
    PARTITION OF inventory_transactions
    FOR VALUES FROM ('2025-01-01') TO ('2025-02-01');

CREATE INDEX idx_invtxn_variant       ON inventory_transactions(variant_id, txn_date DESC);
CREATE INDEX idx_invtxn_ref           ON inventory_transactions(reference_type, reference_id);


CREATE TABLE inventory_reservations (
    reservation_id   UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    variant_id       UUID           NOT NULL,
    warehouse_id     UUID           NOT NULL,
    order_id         UUID,
    cart_id          UUID,
    qty              INTEGER        NOT NULL,
    status           VARCHAR(20)    NOT NULL DEFAULT 'active',
    expires_at       TIMESTAMPTZ    NOT NULL,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now()
);

CREATE INDEX idx_res_variant_wh       ON inventory_reservations(variant_id, warehouse_id, status);
CREATE INDEX idx_res_expires          ON inventory_reservations(expires_at) WHERE status = 'active';
Orders & Cart Domain
Immutable order ledger, partitioned by month, with full state machine
PostgreSQL
-- ============================================================
-- DOMAIN: ORDERS & CART
-- Partitioned by created_at (monthly), immutable after confirm
-- ============================================================

CREATE TABLE carts (
    cart_id          UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id          UUID           REFERENCES users(user_id),
    session_id       TEXT,          -- for guest carts
    currency_code    CHAR(3)        NOT NULL DEFAULT 'INR',
    coupon_code      VARCHAR(50),
    discount_amount  NUMERIC(14,2)  DEFAULT 0,
    notes            TEXT,
    status           VARCHAR(20)    NOT NULL DEFAULT 'active',
    expires_at       TIMESTAMPTZ,
    converted_at     TIMESTAMPTZ,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now(),
    updated_at       TIMESTAMPTZ    NOT NULL DEFAULT now()
);

CREATE INDEX idx_cart_user            ON carts(user_id, status);
CREATE INDEX idx_cart_session         ON carts(session_id) WHERE user_id IS NULL;


CREATE TABLE cart_items (
    cart_item_id     UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    cart_id          UUID           NOT NULL REFERENCES carts(cart_id) ON DELETE CASCADE,
    variant_id       UUID           NOT NULL,
    quantity         INTEGER        NOT NULL CHECK(quantity > 0),
    unit_price       NUMERIC(14,2)  NOT NULL,
    seller_id        UUID           NOT NULL,
    warehouse_id     UUID,
    saved_for_later  BOOLEAN        DEFAULT FALSE,
    added_at         TIMESTAMPTZ    NOT NULL DEFAULT now(),
    UNIQUE(cart_id, variant_id)
);

CREATE INDEX idx_cart_items_cart      ON cart_items(cart_id);


CREATE TABLE orders (
    order_id         UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    order_number     VARCHAR(30)    NOT NULL UNIQUE,  -- ORD-2025-8473682
    user_id          UUID           NOT NULL REFERENCES users(user_id),
    cart_id          UUID,
    status           VARCHAR(30)    NOT NULL DEFAULT 'pending_payment',
                     -- pending_payment | payment_failed | confirmed | processing
                     -- partially_shipped | shipped | delivered | cancelled | refund_initiated | refunded
    payment_status   VARCHAR(30)    NOT NULL DEFAULT 'pending',
    fulfillment_status VARCHAR(30)  NOT NULL DEFAULT 'unfulfilled',

    -- Address snapshot (denormalized for immutability)
    shipping_name    TEXT           NOT NULL,
    shipping_phone   TEXT           NOT NULL,
    shipping_line1   TEXT           NOT NULL,
    shipping_line2   TEXT,
    shipping_city    VARCHAR(100)   NOT NULL,
    shipping_state   VARCHAR(100)   NOT NULL,
    shipping_postal  VARCHAR(20)    NOT NULL,
    shipping_country CHAR(2)        NOT NULL,
    shipping_lat     DECIMAL(9,6),
    shipping_lng     DECIMAL(9,6),

    -- Financials
    currency_code    CHAR(3)        NOT NULL,
    subtotal         NUMERIC(14,2)  NOT NULL,
    discount_amount  NUMERIC(14,2)  NOT NULL DEFAULT 0,
    coupon_code      VARCHAR(50),
    shipping_fee     NUMERIC(14,2)  NOT NULL DEFAULT 0,
    tax_amount       NUMERIC(14,2)  NOT NULL DEFAULT 0,
    total_amount     NUMERIC(14,2)  NOT NULL,
    refunded_amount  NUMERIC(14,2)  NOT NULL DEFAULT 0,

    -- Meta
    source_channel   VARCHAR(30),   -- web | ios | android | api
    ip_address       INET,
    user_agent       TEXT,
    notes            TEXT,
    internal_notes   TEXT,
    metadata         JSONB,

    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now(),
    confirmed_at     TIMESTAMPTZ,
    delivered_at     TIMESTAMPTZ,
    cancelled_at     TIMESTAMPTZ,
    cancel_reason    TEXT
) PARTITION BY RANGE (created_at);

-- Monthly partitions
CREATE TABLE orders_2025_01 PARTITION OF orders
    FOR VALUES FROM ('2025-01-01') TO ('2025-02-01');

CREATE INDEX idx_orders_user          ON orders(user_id, created_at DESC);
CREATE INDEX idx_orders_status        ON orders(status, created_at DESC);
CREATE INDEX idx_orders_number        ON orders(order_number);
CREATE INDEX idx_orders_payment       ON orders(payment_status);


CREATE TABLE order_items (
    order_item_id    UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id         UUID           NOT NULL REFERENCES orders(order_id),
    variant_id       UUID           NOT NULL,
    seller_id        UUID           NOT NULL,
    warehouse_id     UUID,
    sku              VARCHAR(100)   NOT NULL,
    product_title    TEXT           NOT NULL,   -- snapshot
    variant_title    VARCHAR(300),
    image_url        TEXT,
    quantity         INTEGER        NOT NULL,
    unit_price       NUMERIC(14,2)  NOT NULL,
    discount_amount  NUMERIC(14,2)  NOT NULL DEFAULT 0,
    tax_amount       NUMERIC(14,2)  NOT NULL DEFAULT 0,
    total_amount     NUMERIC(14,2)  NOT NULL,
    status           VARCHAR(30)    NOT NULL DEFAULT 'confirmed',
    fulfillment_id   UUID,
    returned_qty     INTEGER        NOT NULL DEFAULT 0,
    refunded_amount  NUMERIC(14,2)  NOT NULL DEFAULT 0
);

CREATE INDEX idx_oitems_order         ON order_items(order_id);
CREATE INDEX idx_oitems_seller        ON order_items(seller_id, status);
CREATE INDEX idx_oitems_variant       ON order_items(variant_id);


CREATE TABLE order_status_history (

Payments Domain
PCI-compliant, multi-gateway, double-entry ledger, fraud signals
PostgreSQL
-- ============================================================
-- DOMAIN: PAYMENTS (PCI DSS Compliant)
-- Double-entry financial ledger, no raw card data stored
-- ============================================================

CREATE TABLE payment_methods (
    pm_id            UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id          UUID           NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    provider         VARCHAR(30)    NOT NULL,  -- stripe | razorpay | paytm | upi | netbanking
    method_type      VARCHAR(30)    NOT NULL,  -- card | upi | wallet | bank_account | bnpl
    provider_pm_id   TEXT           NOT NULL,  -- token from payment gateway
    display_name     VARCHAR(100),             -- "Visa ending in 4242"
    card_brand       VARCHAR(20),
    card_last4       CHAR(4),
    card_exp_month   SMALLINT,
    card_exp_year    SMALLINT,
    upi_id           VARCHAR(100),             -- encrypted
    bank_name        VARCHAR(100),
    is_default       BOOLEAN        DEFAULT FALSE,
    is_verified      BOOLEAN        DEFAULT FALSE,
    created_at       TIMESTAMPTZ    NOT NULL DEFAULT now()
);

CREATE INDEX idx_pm_user              ON payment_methods(user_id);


CREATE TABLE payment_transactions (
    txn_id           UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id         UUID           NOT NULL REFERENCES orders(order_id),
    user_id          UUID           NOT NULL,
    pm_id            UUID           REFERENCES payment_methods(pm_id),
    provider         VARCHAR(30)    NOT NULL,

🗄️
Partitioning Cheatsheet
  • orders → RANGE(created_at) monthly
  • order_items → co-partitioned with orders
  • analytics_events → Cassandra, 90-day TTL
  • notifications → RANGE(created_at) monthly
  • inventory_transactions → RANGE(txn_date) monthly
  • shipment_tracking_events → RANGE(created_at) weekly
  • search_queries_log → RANGE(searched_at) daily
⚡
Critical Index Strategy
  • Partial indexes on active records only (huge selectivity)
  • GIN indexes for full-text search on products & reviews
  • BRIN indexes on time-series tables (orders, events)
  • Composite indexes follow query patterns exactly
  • Index-only scans via INCLUDE clause for hot queries
  • pg_partman for automated partition management
  • Bloom filters for high-cardinality UUID lookups
🗃️
Caching Architecture
  • Redis Cluster — sessions, carts (TTL 30 min), rate limits

PostgreSQL — Sample Queries
-- ============================================================
-- KILLER QUERY 1: User Order History with Seller + Shipment
-- Uses partition pruning + index-only scan
-- ============================================================
SELECT
    o.order_number,
    o.status,
    o.total_amount,
    o.created_at,
    oi.product_title,
    oi.quantity,
    oi.unit_price,
    s.tracking_number,
    s.status       AS shipment_status,
    s.estimated_delivery
FROM orders o
JOIN order_items oi   ON oi.order_id = o.order_id
LEFT JOIN shipments s ON s.order_id = o.order_id
WHERE o.user_id = $1
  AND o.created_at >= '2025-01-01'   -- partition pruning!
ORDER BY o.created_at DESC
LIMIT 20;


-- ============================================================
-- KILLER QUERY 2: Available Inventory per Variant across Warehouses
-- Hits the GENERATED ALWAYS AS column
-- ============================================================
SELECT
    w.name           AS warehouse_name,
    w.city,
    i.qty_on_hand,
    i.qty_reserved,
    i.qty_available,  -- computed column
    i.bin_location
FROM inventory i
JOIN warehouses w ON w.warehouse_id = i.warehouse_id
WHERE i.variant_id = $1
  AND i.qty_available > 0
  AND w.is_active = TRUE
ORDER BY w.geohash;  -- nearest warehouse first (pre-sorted geohash)


-- ============================================================
-- KILLER QUERY 3: Seller Settlement Calculation
-- ============================================================
SELECT
    se.display_name,
    SUM(oi.total_amount)                        AS gross_sales,
    SUM(oi.total_amount * c.commission_pct/100)   AS commission,
    SUM(oi.total_amount) - SUM(oi.total_amount * c.commission_pct/100)
                                                AS net_payout
FROM order_items oi
JOIN sellers se   ON se.seller_id = oi.seller_id
JOIN orders o     ON o.order_id  = oi.order_id
JOIN categories c ON c.category_id = (
    SELECT category_id FROM products
    WHERE product_id = oi.variant_id LIMIT 1
)
WHERE oi.seller_id = $1
  AND o.created_at BETWEEN $2 AND $3
  AND oi.status NOT IN ('cancelled', 'returned')
GROUP BY se.display_name;


-- ============================================================
-- KILLER QUERY 4: Top Products by Revenue (last 30 days)
-- Uses analytics_daily_product pre-aggregation
-- ============================================================
SELECT
    p.title,
    p.avg_rating,
    SUM(adp.revenue)    AS total_revenue,
    SUM(adp.purchases)  AS total_orders,
    AVG(adp.conversion_rate) * 100  AS avg_cvr_pct
FROM analytics_daily_product adp
JOIN products p ON p.product_id = adp.product_id
WHERE adp.stat_date >= CURRENT_DATE - 30
GROUP BY p.product_id, p.title, p.avg_rating
ORDER BY total_revenue DESC
LIMIT 50;
